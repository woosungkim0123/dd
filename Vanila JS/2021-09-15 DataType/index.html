자바스크립트 데이터 타입
Vanila / Javascript
2021.09.15.

들어가기 전

특정 데이터에 접근 할 수 있다면 특정 데이터를 변경하는 것도 가능하다

원시 타입(Primitive type)

문자열(String), 숫자(Number), Boolean(True / False), undefined, null

원시타입의 변수들은 데이터 복사가 일어날 때 메모리 공간을 새로 확보하여 독립적인 값을 저장한다

참조 타입

함수(function)

함수란 반복해서 사용할 수 있는 코드 뭉치

function sum(num1, num2) {
  // num1, num2 매개변수라고 한다
  // 실행할 코드
}

sum(10, 20) // 함수로 전달하는 값을 인자 or 인수 라고 한다

// 인자를 매개변수보다 적게 넣는다면?
function area(width, height) {
  console.log(width) // 10
  console.log(height) // undefined
}

area(10)

자바스크립트의 모든 데이터타입은 매개변수로 전달 가능하다

return

함수가 함수 밖과 소통하는 방법


함수 밖에서 결과 값을 원할때는 return


return을 하면 함수는 작동을 멈추고 결과값을 리턴하고 끝냄

function sum(num1, num2) {
  return num1 + num2
}
let result = sum(10, 20)

console.log(result) // 30 출력

배열(Array)

다양한 과일정보를 담고 있는 변수가 있다. 근데 종류가 너무 많아서 따로 만들면 비효율적

// 비효율
const banana = '바나나'
const apple = '사과'
const melon = '멜론'

// 배열 사용
const fruit = ['바나나', '사과', '멜론']

// 0, 1, 2는 배열 안에 있는 데이터(바나나, 사과, 멜론)의 데이터 좌표값(index)이다.
// length는 배열 안에 들어가있는 데이터 갯수

console.log(fruit[0]) // 바나나 출력
console.log(fruit.length) // 3 출력

배열을 사용하게 되면 유사한 성격을 지닌 데이터를 하나의 변수 안에 담아낼 수 있다. (일종의 바구니)

객체(Object)

여러개의 데이터를 하나의 변수 안에 전부다 넣을 때 사용

const warriorA = {
  // hp가 key 그리고 key 값에 대응하는 데이터 값 즉, 300을 value 라고 부른다
  hp: 300,
  mp: 200,
  q: '십자베기',
  w: '십일자베기',
  패시브: ['강인한 체력', '비스트'],
}

console.log(warriorA.hp) // 300 출력
warriorA.hp = 400
console.log(warriorA.hp) // 400 출력
console.log(warriorA['hp']) // 400 출력

warriorA가 가리키고 있는 메모리의 포인터는 const로 잠겨있어서 다른 object로 할당이 불가하지만 warriorA 안에 hp나 mp등의 변수가 존재한다.


warriorA.hp 이런식으로 각각 포인트가 가리키고 있는 메모리에 다른 값으로 할당이 가능하다.


객체 안에 함수도 만들 수 있다

// 객체 안에서 만들어진 함수를 메서드라 한다.
const warriorA = {
  hp: 300,
  sum: function (num1, num2) {
    return num1 + num2
  },
}

console.log(warriorA.sum(10, 20)) // 30출력

우리가 계속 쓰고 있는 console.log도 console 이라는 object에 있는 log 메서드를 사용한 것

원시 vs 참조

원시타입

let str1 = 'Hello world'
let str2 = str1
// str1의 값이 그대로 str2에 할당이 된다
str1 = 'Nice'
console.log(str1) // "Nice"
console.log(str2) // "Hello world"

// str1변수의 데이터를 'Nice'로 수정하면 str1에만 변경되고 str2는 변경되지 않는다

원시타입은 원본데이터를 수정했다고해서 복사본이 바뀌지 않는다.(반대도 마찬가지)


원본과 복사본 데이터 개념이 들어가 있는 데이터타입이 원시타입이라고 할 수 있다.

참조타입

let obj1 = { name: 'Woosung' }
let obj2 = obj1
// 데이터를 변경하게되면 둘다 바뀌게 된다 (반대도 마찬가지)
obj1.name = 'Kim'

console.log(obj1) // "Kim"
console.log(obj2) // "Kim"

참조타입은 직접적으로 데이터를 변수 안에다가 넣어주는 것이 아니고 데이터를 메모리라는 저장소에 넣어둔다.


즉, obj1은 데이터를 가리키는 것이 아니라 데이터가 저장되어있는 주소를 말하는 것이다.


obj2도 obj1이 가리키는 있는 주소를 가리키도록 만들었기 때문에 주소를 공유하고 있는 것이다.


정리하면 원시 타입은 value로 값이 저장되고 object는 object를 가르키는 참조(reference)가 메모리에 저장 된다
