BrowserRouter vs HashRouter

두개의 차이점은 URL 생김새에 있다


BrowserRouter에서 url은 보통 웹사이트처럼 생겼으나 HashRouter는 뒤에 #이 붙는다

// BrowserRouter
localhost:3000/movie

// HashRouter
localhost:3000/#/movie

대부분은 BrowserRouter를 사용함


Link

동적(다이나믹) url을 지원함

// :id는 변수처럼 사용가능
function App() {
  return (
    <Router>
      <Routes>
        <Route path="/movie/:id" element={<Detail />} />
      </Routes>
    </Router>
  )
}

// Detail
// useParams으로 값을 받을 수 있다
function Detail() {
  const { id } = useParams()
  const getMoives = async () => {
    const json = await (
      await fetch(`https://yts.mx/api/v2/movie_details.json?movie_id=${id}`)
    ).json()

    useEffect(() => {
      getMoives()
    }, [])
  }
  return (
    <div>
      <h1>Detail</h1>
    </div>
  )
}
