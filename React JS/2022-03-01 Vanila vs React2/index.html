Vanila JS vs React JS (State 편)
React / Javascript
2022.03.01.

Vanila로 만든 클릭시 숫자가 변하는 부분을 React로 구현해보자

좋은 방법과 좋지않는 방법이 있는데 좋지 않는 방법부터 알아보자

좋지 않은 방법(React)

// Vanila JS
const button = document.getElementById('btn')
const span = document.querySelector('span')
let counter = 0

function 클릭카운트증가() {
  counter = counter + 1
  span.innerText = '총 클릭수 : ' + counter
}

button.addEventListener('click', 클릭카운트증가)
// React JS 좋지 않은 방법

// 변수를 JSX에 전달하는 방법 : {변수명}
let 카운트 = 0

function 클릭카운트증가() {
  카운트 = 카운트 + 1
}
const Container = () => (
  <div>
    <span>총 클릭수 : {카운트}</span>
    <button onClick={클릭카운트증가}></button>
  </div>
)
ReactDOM.render(<Container />, root)

// 이 코드는 제대로 작동되지 않음

컴포넌트를 단 한번만 렌더링하기 때문에 counter는 증가하지만 UI가 업데이트(리렌더링) 되지 않아 반영되지 않음

그러면 우리가 원하는대로 ReRender 시켜보자!

// 클릭할때마다 함수가 실행되면서 새로 렌더링
// Rendering : 바꾸어 보여주는 작업

let 카운트 = 0

function 클릭카운트증가() {
  카운트 = 카운트 + 1
  렌더링()
}

function 렌더링() {
  ReactDOM.render(<Container />, root)
}

const Container = () => (
  <div>
    <span>총 클릭수 : {카운트}</span> // 변수를 JSX에 전달하는 방법
    <button onClick={클릭카운트증가}>클릭해줘</button>
  </div>
)
렌더링()
  1. 어플리케이션이 실행될 때 카운터는 0,
  2. ReactDOM이 Container를 root에 렌더링 시킨다 (카운터 : 0)
  3. root라는 빈 div안에 Container 컴포넌트의 내용물들이 추가되기 시작(span, button, 이벤트)
  4. 버튼을 클릭하게 되면 클릭카운트증가 함수를 호출하게 되고 카운터를 증가 (카운터 : 1)
  5. 그리고 렌더링 함수를 호출, Container가 다시 렌더링된다.(카운터 1이 화면에 보임)
<!-- Vanila -->
<body>
  <!-- 렌더링 -->
  <span>
    <!-- 렌더링 -->
    총 클릭수 :
    <!-- 렌더링 -->
    카운트
    <!-- 렌더링 -->
  </span>
</body>

<!-- React -->
<body>
  <div>
    <span>
      총 클릭수 : 카운트
      <!-- 이곳만 렌더링 -->
    </span>
    <button>클릭해줘</button>
  </div>
</body>

React.js는 이전에 렌더링된 컴포넌트를 알고 다음에 렌더링 될 컴포넌트에서 다른 부분만 파악한다

좋은 방법(React)

안좋은 방법에서는 데이터가 바뀔때마다 렌더링 함수를 호출하는 걸 잊어서는 안된다.

// React.js 어플 내에서 데이터를 보관하고 자동으로 리렌더링을 일으킬 수 있는 방법
// Container은 App으로 변경(관습)

const root = document.getElementById('root')

function App() {
  const data = React.useState(0)
  console.log(data) // [0, 함수]
  // 첫번째가 data, 두번째가 data를 변경할 때 사용하는 함수

  return (
    <div>
      <span>총 클릭수 : {카운트}</span>
      <button onClick={클릭카운트증가}>클릭해줘</button>
    </div>
  )
}
ReactDOM.render(<App />, root)
// 안좋은 방법
let 카운트 = 0 // data 초기값

function 클릭카운트증가() {
  // data값을 변경하는 함수
  카운트 = 카운트 + 1
  렌더링()
}

// 좋은 방법
const data = React.useState(0) // [0, 함수]

const 카운터 = data[0]
const 카운터수정 = data[1]

// 조금 더 깔끔하게 표현
const [카운터, 카운터수정] = React.useState(0)

State?

state는 기본적으로 object이며, 컴퍼넌트 자신이 가지고 있는 속성 값

간단히 말해 리렌더링을 통해 자주 변경될 것 같은 값을 변수가 아닌 state에 저장해서 관리함

카운트 값(state)을 변경해보자

// 왜 변경함수를 사용해야 하는가?

...

let [카운트, 카운트수정] = React.useState(0)
const 클릭카운트증가 = () => {
	카운트 = 카운트 + 1
}
return(
	<div>
		<span>총 클릭수 : { 카운트 }</span>
		<button onClick={ 클릭카운트증가 }>클릭해줘</button>
	</div>
)
...

// 카운트가 증가하나 리렌더링이 되지않아서 UI가 업데이트 되지않음

const 클릭카운트증가 = () => {
	카운트 = 카운트 + 1
	ReactDOM.render(<App />, root) // 리렌더링이 필요한 곳마다 넣어줘야함
}

// 이것 떄문에 변경함수가 존재

const [카운트, 카운트수정] = React.useState(0)
// 보통 state의 이름이 counter면 변경함수의 이름은 setCounter
const 클릭카운트증가 = () => {
	카운트수정( 카운트 + 1 )
	// 변경함수는 안에 들어있는 값으로 업데이트하고 리렌더링을 시켜준다
	// 값을 바꾸고 리렌더링, 이 모든 것이 변경함수 안에서 일어난다
}
return(
	<div>
		<span>총 클릭수 : { 카운트 }</span>
		<button onClick={ 클릭카운트증가 }>클릭해줘</button>
	</div>
)

변경함수를 이용해서 state를 변경할 때마다 컴포넌트 전체가 새로운 값을 가지고 재생성된다.

이때 전부 바뀌는 것이 아닌 변경되는 부분만 바뀌게 된다
