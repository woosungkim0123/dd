개념

이전에는 변수에 데이터를 저장하고 데이터를 바인딩 했다.


변수말고 데이터를 저장하는 또 하나의 방법이 바로 state 이다.


변수로 만들면 데이터가 변경이 되면 재랜더링이 바로바로 되지않고 새로고침이 되어야 재랜더링이 됨


state로 만들어놓은 데이터들은 state가 조작이 되더라도 새로고침없이도 재랜더링을 해준다 (새로고침 없이 HTML이 바뀜)

변경함수

state는 변수처럼 값만 집어넣는 것이 아닌 그 값을 변경해줄 함수까지 같이 정의하게 된다

// useState
let [글제목, 글제목변경] = useState('제목1')

// state 변경
// 변경함수는 소괄호 안에 있는 데이터로 state 데이터를 싹다 바꿔주는 방식
글제목변경('변경된 제목')
let [글제목, 글제목변경] = useState(['제목1', '제목2', '제목3'])

// 원본 state데이터는 수정이 불가능하다
// 복사본을 만들어서 수정하는 방식
const 제목바꾸기 = () => {
  let 개발자방법 = [...글제목]
  개발자방법[0] = '변경된 제목1'
  글제목변경(개발자방법)
}

예시

function App() {
  let [글제목, 글제목변경] = useState('남자 코트 추천')

  return (
    <div className="App">
      <div className="black-nav">
        <div>개발 Blog</div>
      </div>
      <div className="list">
        <h3>{글제목}</h3>
        <p>10월 5일 발행</p>
        <hr />
      </div>
    </div>
  )
}
